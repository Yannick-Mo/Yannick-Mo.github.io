def quick_sort(arr):
    """快速排序算法实现"""
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quick_sort(left) + middle + quick_sort(right)

# 测试代码
if __name__ == "__main__":
    data = [3, 6, 8, 10, 1, 2, 1]
    print(f"原始数组: {data}")
    print(f"排序结果: {quick_sort(data)}")

// 异步函数示例
async function fetchData(url) {
    try {
        const response = await fetch(url);
        const data = await response.json();
        
        // 使用可选链操作符
        const result = data?.items?.map(item => ({
            id: item.id,
            name: item.name?.toUpperCase() || 'Unknown',
            value: item.value ?? 0 // 空值合并运算符
        }));
        
        return result;
    } catch (error) {
        console.error('获取数据失败:', error);
        throw new Error(`请求失败: ${error.message}`);
    }
}

// 使用示例
fetchData('https://api.example.com/data')
    .then(data => console.log(data))
    .catch(error => console.error(error));

4. 数学公式测试

行内公式

爱因斯坦的质能方程：$E = mc^2$，其中 $E$ 表示能量，$m$ 表示质量，$c$ 表示光速。

独立公式

二次方程求根公式：

$$ x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a} $$

微积分公式

牛顿-莱布尼茨公式：

$$ \int_a^b f(x) dx = F(b) - F(a) $$

其中 $F'(x) = f(x)$。

极限公式

重要极限：

$$ \lim_{x \to 0} \frac{\sin x}{x} = 1 $$

概率论公式

贝叶斯定理：

$$ P(A|B) = \frac{P(B|A)P(A)}{P(B)} $$

12. 数学公式进阶测试

傅里叶变换

$$ \mathcal{F}\{f(t)\} = F(\omega) = \int_{-\infty}^{\infty} f(t) e^{-j\omega t} dt $$

微分方程

一阶线性微分方程：

$$ \frac{dy}{dx} + P(x)y = Q(x) $$

其通解为：

$$ y = e^{-\int P(x) dx} \left( \int Q(x) e^{\int P(x) dx} dx + C \right) $$

注意事项

  1. 图片路径：实际使用时，请确保图片路径正确，可以将图片放在 static/images/ 目录下

  3. Mermaid 图表：确保在 front matter 中设置 mermaid: true
  4. 视频/音频：需要将媒体文件放在 static 目录下
