1. 概述：什么是IPC消息队列

消息队列是操作系统内核维护的一个消息链表，允许进程以"消息"为单位进行数据交换。每条消息通常包含消息类型和消息内容，相比无名管道（字节流），消息队列支持结构化数据传输和按类型过滤接收。

核心优势

  • 解耦：发送方与接收方无需同时运行
  • 异步：发送后即可继续执行，接收方延迟处理
  • 缓冲：队列可在一定范围内暂存多条消息

与其它 IPC 的对比

机制 适用场景 是否跨主机 数据形式
管道 简单父子进程流 否 字节流
消息队列 结构化消息，异步通信 否 带类型消息
共享内存 极高频大数据传输 否 内存块（需同步）
Socket 网络通信 是 字节流/报文

两大主流标准

特性 System V 消息队列 POSIX 消息队列
标识 整型key (ftok) 字符串名 (/myqueue)
消息组织 通过长整型类型精确选择 通过优先级（数字越大越优先）
异步通知 不支持 支持 mq_notify
可移植性 所有Unix类系统 Linux、部分BSD，macOS不支持
推荐场景 遗留系统、需要复杂类型过滤 新Linux项目、高优先级需求

#ifndef COMMON_SV_H
#define COMMON_SV_H

#include <sys/ipc.h>
#include <sys/msg.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

struct perf_msg {
    long mtype;               // 消息类型（必须第一个）
    int  employee_id;
    char name[32];
    float score;
    unsigned int level;
    char comment[128];
};

static inline int send_perf_msg(int msqid, const struct perf_msg *msg) {
    // 注意：msgsz = 总大小 - sizeof(long)
    return msgsnd(msqid, msg, sizeof(*msg) - sizeof(long), 0);
}

static inline int recv_perf_msg(int msqid, long type, struct perf_msg *msg) {
    return msgrcv(msqid, msg, sizeof(*msg) - sizeof(long), type, 0);
}

#endif

int main() {
    key_t key = ftok(".", 'p');
    int msqid = msgget(key, IPC_CREAT | 0666);

    struct perf_msg msg = {
        .mtype = 1,
        .employee_id = 1001,
        .score = 92.5,
        .level = 4
    };
    strcpy(msg.name, "张三");
    strcpy(msg.comment, "技术能力强");

    send_perf_msg(msqid, &msg);
    return 0;
}

    struct perf_msg msg;

    recv_perf_msg(msqid, 1, &msg);

}

#ifndef COMMON_POSIX_H
#define COMMON_POSIX_H

#include <mqueue.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define QUEUE_NAME "/sensor_queue"
#define MAX_SENSORS 5

typedef struct {
    int sensor_id;
    double temperature;
    double humidity;
    long timestamp;
} sensor_reading_t;

typedef struct {
    int count;
    char unit[8];
    sensor_reading_t readings[MAX_SENSORS];
} sensor_batch_t;

static inline mqd_t open_sensor_queue(int oflag) {
    struct mq_attr attr = {
        .mq_maxmsg = 10,
        .mq_msgsize = sizeof(sensor_batch_t),
        .mq_flags = 0
    };
    return mq_open(QUEUE_NAME, oflag, 0666, &attr);
}
#endif

#include "common_posix.h"

int main() {
    mqd_t mq = open_sensor_queue(O_CREAT | O_WRONLY);
    sensor_batch_t batch = {
        .count = 3,
        .unit = "metric"
    };
    for (int i = 0; i < batch.count; i++) {
        batch.readings[i] = (sensor_reading_t){
            .sensor_id = 100 + i,
            .temperature = 20.5 + i * 1.5,
            .humidity = 45.0 + i * 2.0,
            .timestamp = time(NULL)
        };
    }
    mq_send(mq, (const char*)&batch, sizeof(batch), 1);
    mq_close(mq);
    return 0;
}

int main() {
    mqd_t mq = open_sensor_queue(O_RDONLY);
    sensor_batch_t batch;
    unsigned int prio;
    mq_receive(mq, (char*)&batch, sizeof(batch), &prio);
    printf("收到 %d 个传感器数据\n", batch.count);
    mq_close(mq);
    mq_unlink(QUEUE_NAME);
    return 0;
}

3. 必要补充知识

3.1 系统限制与调优

类型 限制项 查看/设置方式 Linux 典型默认值
System V 单条消息最大字节数 kernel.msgmax 8192 字节
System V 单个队列总字节数 kernel.msgmnb 16384 字节
System V 系统最大队列数 kernel.msgmni 16
POSIX 系统最大队列数 /proc/sys/fs/mqueue/queues_max 256
POSIX 单条消息最大字节数 /proc/sys/fs/mqueue/msgsize_max 8192 字节
POSIX 单用户总字节上限 ulimit -q 819200 字节 (800KB)

3.2 常用管理命令

System V
  • ipcs -q ：查看所有消息队列
  • ipcrm -q <msqid>：删除指定队列
POSIX
  • ls /dev/mqueue ：查看 POSIX 队列（需挂载 mqueue 文件系统）
  • cat /dev/mqueue/<队列名> ：查看队列属性

3.3 自定义数据结构的注意事项

  • System V：msgsnd/msgrcv 的 msgsz 参数必须是 结构体总大小减去 sizeof(long)，因为 mtype 由内核单独处理。
  • POSIX：完全自由，发送 sizeof(struct) 字节即可，但需保证接收端缓冲区足够大。
  • 跨平台/跨架构：不同 CPU 或编译器的结构体对齐和字节序可能不同。对于严格跨环境通信，建议使用序列化（如 JSON、Protobuf、MessagePack）或显式处理字节序（htonl/ntohl）。

char msg[1024];

char buf[1024];

更通用的方案：先发送固定长度的头部（含后续数据长度），再发送数据体（分段发送或整体发送）。

3.5 同步与异步通知

  • System V：无法直接异步通知，只能通过轮询或结合信号（如 SIGIO，但实现复杂）。
  • POSIX：mq_notify() 可在新消息到达时发送信号或启动线程，避免阻塞或轮询。

// 注册异步通知（信号方式）
struct sigevent sev;
sev.sigev_notify = SIGEV_SIGNAL;
sev.sigev_signo = SIGUSR1;
mq_notify(mq, &sev);

3.6 资源清理

  • System V：进程退出时队列不会自动删除，必须显式调用 msgctl(msqid, IPC_RMID, NULL) 或使用 ipcrm 命令。否则队列会一直占用内核资源。
  • POSIX：引用计数式管理。所有进程关闭描述符（mq_close）后，若调用了 mq_unlink，则队列被删除；若未调用 unlink，队列会持久化。

总结

  • 消息队列是单机多进程间传递结构化数据的高效 IPC 方式，具有解耦、异步、缓冲的优势。
  • System V 和 POSIX 是两种主流实现，各有侧重：System V 类型过滤灵活，POSIX 接口现代且支持优先级和通知。
  • 实际使用中，强烈推荐优先考虑 POSIX 消息队列（除非需要兼容 macOS 或遗留系统）。
  • 自定义数据结构十分自由，只需注意 System V 的 msgsz 长度计算，以及跨平台时的序列化问题。
  • 务必了解系统默认限制，并根据需求调整内核参数；同时做好资源清理，避免队列泄漏。
